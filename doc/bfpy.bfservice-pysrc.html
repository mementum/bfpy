192          ''' 
193          Invocation of a service in a generic manner. 
194   
195          It is in charge of deciding the final endPoint (global or exchange, 
196          and if exchange, which one from the parameters), get a reference 
197          to the service method from the instance, the request to be passed 
198          and filling in the request values (with the default and provided 
199          kwargs) 
200   
201          Once the response is received, it is passed through the defined 
202          post processing callables and then returned 
203   
204          @param instance: object whose method is being invoked 
205          @type instance: object 
206          @param args: exchange services will receive the exchangeId there 
207          @type args: tuple 
208          @param kwargs: names and values of args for the service request 
209          @type kwargs: dict 
210          ''' 
211          # Calls to Exchanges need the specific exchange endPoint 
212          # passed to the functions as 1st unnamed arg 
213          try: 
214              endPoint = self.endPoint if self.endPoint <= 0 else args[0] 
215          except Exception, e: 
216              raise bferror.BfPythonError('Forgot to pass ExchangeId!', e, str(e), e.args) 
217   
218          # Get the service from the appropriate endpoint 
219          service = instance.getService(endPoint, self.serviceName) 
220   
221          # Get the service request argument from the appropriate endpoint 
222          request = instance.getRequest(endPoint, self.requestName, self.apiHeader) 
223   
224          # Make a copy dictionary of args with default values 
225          requestArgs = self.kwargs.copy() 
226          # Update the dictionary with new args/updated args 
227          requestArgs.update(kwargs) 
228   
229          # Pre-process 
230          if instance.preProcess: 
231              for preProc in self.preProc: 
232                  preProc(request, requestArgs, 
233                          instance=instance, exchangeId=endPoint, 
234                          methodArgs=kwargs) 
235   
236          # Fill in the defined argument values in the service request 
237          for name, value in requestArgs.iteritems(): 
238              if not name.startswith('_'): 
239                  setattr(request, name, value) 
240   
241          # Calculate the data charges weight 
242          weight = abs(self.weight) 
243          if self.weight < 0: 
244              if hasattr(request, 'marketId') and not request.marketId: 
245                  # Some calls using marketId '0' cost 5 times 
246                  weight *= 5 
247               
248          # Add the weight to the instance - the call will block if needed 
249          instance.addDataWeight(endPoint, weight) 
250   
251          # Execute the call and fetch the response 
252          response = instance.invoke(self.methodName, service, 
253                                     request, self.skipErrorCodes) 
254   
255          # Post-Process the response if needed 
256          if instance.postProcess: 
257              for postProc in self.postProc: 
258                  postProc(response, instance=instance, exchangeId=endPoint, 
259                           request=request, requestArgs=requestArgs) 
260   
261          return response
