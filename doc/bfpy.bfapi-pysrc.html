350          ''' 
351          Invokes a service with a given request and with a list of errors 
352          that do not generate exceptions if returned 
353   
354          @param methodName: method name that has been used to invoke 
355          @type methodName: str 
356          @param service: name of the object to be retrieved 
357          @type service: method generated by suds from the WSDL definition 
358          @param skipErrorCodes: error codes that will not generate exceptions 
359          @type skipErrorCodes: list 
360   
361          @return: the response object processed by suds 
362          @rtype: object generated by suds from the WSDL definitions 
363   
364          @raise BfHttpError: if the server does not answer with 200 OK 
365          @raise BfNetworkError: on (potential) network errors 
366          @raise Exception: on other errors, due to the fact that suds does not 
367                            generate all exceptions from a common base 
368          @raise BfApiError: on specific API errors 
369          @raise BfServiceError: on service errors (unless specified in skipErrorCodes) 
370          ''' 
371          try: 
372              response = service(request) 
373          except suds.WebFault, e: 
374              raise bferror.BfHttpError(methodName, e, str(e), e.fault, e.document) 
375          except (SocketError, bftransport.TransportException), e: 
376              # Summarise (all potential) network errors into a generic notification 
377              raise bferror.BfNetworkError(methodName, e, str(e), e.args) 
378          except Exception, e: 
379              # Catch all - needed because suds doesn't have a root class Exception 
380              # and also produces some standard exceptions 
381              raise bferror.BfError(methodName, e, str(e), e.args) 
382   
383          # Analyze API errorCodes 
384          if response.header.errorCode != 'OK': 
385              raise bferror.BfApiError(methodName, response, str(response), response.header.errorCode) 
386   
387          # Analyze Service errorCodes - keepAlive (at least) has no "errorCode" field 
388          if hasattr(response, 'errorCode'): 
389              if response.errorCode != 'OK' and response.errorCode not in skipErrorCodes: 
390                  raise bferror.BfServiceError(methodName, response, str(response), response.errorCode) 
391   
392          # Save the latest session token 
393          self.sessionToken = response.header.sessionToken 
394   
395          return response
