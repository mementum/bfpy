300          ''' 
301          Invokes a method by name, passing the supplied request argument by using specific 
302          suds wsdl client (specified by bfServiceId) 
303   
304          The method checks for the existence of the method before invocation and raises 
305          different exceptions in case errors happen. 
306   
307          If the call to the Betfair API succeeds but a high level errorCode is returned, 
308          a list of goodErrorCodes is checked before raising the exception. This is so, 
309          because the Betfair API may have errors (returning INVALID_MARKET instead of 
310          MARKET_CLOSED in getProfitAndLoss) or because an error like 
311          INVALID_LOCALE_DEFAULTING_TO_ENGLISH may be returned, which still provides the 
312          requested info even if in a different language. 
313   
314          Suds raises some plain exceptions and its exceptions do not have a base 
315          "SudsError" class. Therefore a catch-all "Exception" is implemented to try a 
316          good error reporting to the calling application 
317   
318          @param bfServiceId: id of the suds wsdl client to use 
319          @type bfServiceId: int 
320          @param requestName: service to invoke 
321          @type requestName: str 
322          @param requestArg: object to be passed as param of the request 
323          @type requestArg: suds object 
324          @param goodErrorCodes: list of errorCodes to skip 
325          @type goodErrorCodes: list 
326   
327          @returns: Betfair API answer 
328          @rtype: suds object 
329   
330          @raise BfPythonError: if the method is not found in the suds client 
331          @raise BfHttpError: if suds raises a WebFault 
332          @raise BfNetworkError: on potential network errors 
333          @raise BfError: on generic Exceptions 
334          @raise BfApiError: if the API replies with a ApiHeader error 
335          @raise BfServiceError: if the API replies with a Service error and 
336                                 the error is not overseen by goodErrorCodes 
337          ''' 
338          if hasattr(self.bfClient[bfServiceId].service, requestName): 
339              method = getattr(self.bfClient[bfServiceId].service, requestName) 
340          else: 
341              raise bferror.BfPythonError(requestName, self.bfClient[bfServiceId], 'Method Not found') 
342   
343          try: 
344              response = method(requestArg) 
345          except suds.WebFault, e: 
346              raise bferror.BfHttpError(requestName, e, str(e), e.fault, e.document) 
347          except (SocketError, bftransport.TransportException), e: 
348              # Summarise network errors into a generic notification 
349              raise bferror.BfNetworkError(requestName, e, str(e), e.args) 
350          except Exception, e: 
351              # Catch all - needed because suds is inconsistent with error handling 
352              raise bferror.BfError(requestName, e, str(e), e.args) 
353   
354          # Response is returned 
355   
356          # First check API errors 
357          if response.header.errorCode != 'OK': 
358              raise bferror.BfApiError(requestName, response, str(response), response.header.errorCode) 
359   
360          # Some services don't return errorCode, check before error checking 
361          if hasattr(response, 'errorCode'): 
362              # if errorCode is present check it is ok or an accepted errorCode 
363              if not (response.errorCode == 'OK' or (goodErrorCodes and response.errorCode in goodErrorCodes)): 
364                  raise bferror.BfServiceError(requestName, response, str(response), response.errorCode) 
365   
366          # Save the "new" session token 
367          self.sessionToken = response.header.sessionToken 
368          return response
